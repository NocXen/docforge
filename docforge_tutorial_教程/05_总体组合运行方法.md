# DocForge 教程 05：总体组合运行方法

本教程介绍如何将实例、工作流和插件组合起来，完成一个完整的文档处理任务。这是 DocForge 的核心使用方式。

## 1. 总体流程概述

一个完整的 DocForge 任务包含以下组件：
1. **实例**：Python 脚本，调用 CoreAPI 协调整个流程
2. **工作流**：JSON 文件，定义处理步骤和插件配置
3. **插件**：Python 类，执行具体的提取、转换、替换等操作
4. **输入文件**：Excel、Word 等待处理的源文件
5. **模板文件**：用于生成输出的模板
6. **输出目录**：生成文件的存放位置

## 2. 准备工作

### 2.1 目录结构
一个典型的项目目录结构如下：
```
my_project/
├── data/                    # 输入数据
│   ├── input1.xlsx
│   └── input2.xlsx
├── templates/               # 模板文件
│   ├── template1.docx
│   └── template2.docx
├── plugins/                 # 自定义插件（可选）
│   └── my_extractor.py
├── output/                  # 输出目录
├── workflow.json            # 工作流定义
├── run_instance.py          # 实例脚本
└── requirements.txt         # 依赖列表
```

## 4. 编写实例脚本

### 4.1 基本实例脚本
创建 `run_instance.py`：
```python
"""
run_instance.py
DocForge 实例脚本
"""

import sys
import json
from pathlib import Path

# 添加项目根目录到 Python 路径
project_root = Path(__file__).parent
sys.path.insert(0, str(project_root))

from docforge.api.core_api import CoreAPI

def main():
    print("=" * 60)
    print("DocForge 文档处理任务")
    print("=" * 60)
    
    # 1. 初始化
    api = CoreAPI()
    if not api.initialize():
        print("❌ 框架初始化失败")
        return 1
    
    print("✅ 框架初始化成功")
    
    # 2. 创建项目
    project_dir = project_root / "project"
    if not api.create_project("文档处理项目", str(project_dir)):
        print("❌ 创建项目失败")
        api.shutdown()
        return 1
    
    print("✅ 项目创建成功")
    
    # 3. 添加输入文件
    data_dir = project_root / "data"
    input_files = list(data_dir.glob("*.xlsx")) + list(data_dir.glob("*.xls"))
    
    if not input_files:
        print(f"❌ 没有找到输入文件: {data_dir}")
        api.shutdown()
        return 1
    
    if not api.add_input_files([str(f) for f in input_files]):
        print("❌ 添加输入文件失败")
        api.shutdown()
        return 1
    
    print(f"✅ 添加了 {len(input_files)} 个输入文件")
    
    # 4. 添加模板文件
    template_dir = project_root / "templates"
    template_files = list(template_dir.glob("*.docx"))
    
    if not template_files:
        print(f"❌ 没有找到模板文件: {template_dir}")
        api.shutdown()
        return 1
    
    if not api.add_template_files([str(f) for f in template_files]):
        print("❌ 添加模板文件失败")
        api.shutdown()
        return 1
    
    print(f"✅ 添加了 {len(template_files)} 个模板文件")
    
    # 5. 设置输出目录
    output_dir = project_root / "output"
    output_dir.mkdir(exist_ok=True)
    
    if not api.set_output_directory(str(output_dir)):
        print("❌ 设置输出目录失败")
        api.shutdown()
        return 1
    
    print(f"✅ 输出目录设置为: {output_dir}")
    
    # 6. 加载工作流
    workflow_path = project_root / "workflow.json"
    if not workflow_path.exists():
        print(f"❌ 工作流文件不存在: {workflow_path}")
        api.shutdown()
        return 1
    
    if not api.load_workflow(str(workflow_path)):
        print("❌ 加载工作流失败")
        api.shutdown()
        return 1
    
    print("✅ 工作流加载成功")
    
    # 7. 加载插件（从默认插件目录）
    plugin_dir = project_root / "plugins"
    if plugin_dir.exists():
        api.plugin_manager.load_all_plugins(str(plugin_dir))
        print(f"✅ 加载了自定义插件目录: {plugin_dir}")
    
    # 8. 执行工作流
    print("\n" + "-" * 60)
    print("开始执行工作流...")
    print("-" * 60)
    
    def progress_callback(progress, message):
        print(f"[{progress*100:5.1f}%] {message}")
    
    try:
        # 获取工作流名称
        with open(workflow_path, 'r', encoding='utf-8') as f:
            workflow_config = json.load(f)
        
        workflow_name = workflow_config.get('name', 'default_workflow')
        
        # 执行工作流
        result = api.execute_workflow(
            workflow_name=workflow_name,
            progress_callback=progress_callback
        )
        
        print("-" * 60)
        
        # 9. 处理结果
        if result.success:
            print(f"\n✅ 工作流执行成功!")
            print(f"   生成文件数: {len(result.output_files)}")
            if result.output_files:
                print("   输出文件列表:")
                for file in result.output_files:
                    print(f"     - {file}")
        else:
            print(f"\n❌ 工作流执行失败")
            if result.errors:
                print("   错误信息:")
                for error in result.errors:
                    print(f"     - {error}")
        
        # 10. 显示执行信息
        print(f"\n执行时间: {result.execution_time:.2f} 秒")
        
    except Exception as e:
        print(f"\n❌ 执行过程中出错: {e}")
        import traceback
        traceback.print_exc()
        api.shutdown()
        return 1
    
    # 11. 关闭框架
    api.shutdown()
    print("\n" + "=" * 60)
    print("任务完成!")
    print("=" * 60)
    
    return 0

if __name__ == "__main__":
    sys.exit(main())
```

### 4.2 带配置的实例脚本
支持配置文件的实例脚本：
```python
"""
run_instance_config.py
支持配置的实例脚本
"""

import sys
import json
from pathlib import Path
from docforge.api.core_api import CoreAPI

def load_config(config_path):
    """加载配置文件"""
    with open(config_path, 'r', encoding='utf-8') as f:
        return json.load(f)

def main():
    # 加载配置
    config_path = "config.json"
    if not config_path.exists():
        print(f"配置文件不存在: {config_path}")
        return 1
    
    config = load_config(config_path)
    
    # 初始化
    api = CoreAPI()
    if not api.initialize():
        return 1
    
    # 使用配置
    project_config = config.get("project", {})
    api.create_project(
        project_config.get("name", "默认项目"),
        project_config.get("path", "./project")
    )
    
    # 添加文件
    files_config = config.get("files", {})
    api.add_input_files(files_config.get("input_files", []))
    api.add_template_files(files_config.get("template_files", []))
    api.set_output_directory(files_config.get("output_dir", "./output"))
    
    # 加载工作流
    workflow_config = config.get("workflow", {})
    api.load_workflow(workflow_config.get("path", "workflow.json"))
    
    # 执行
    result = api.execute_workflow(workflow_config.get("name", "default_workflow"))
    
    # 输出结果
    if result.success:
        print(f"成功生成 {len(result.output_files)} 个文件")
    else:
        print(f"执行失败: {result.errors}")
    
    api.shutdown()
    return 0

if __name__ == "__main__":
    sys.exit(main())
```

## 6. 高级用法

### 6.1 错误恢复
在实例中添加错误恢复机制：
```python
def main_with_retry():
    """带重试的实例"""
    max_retries = 3
    
    for attempt in range(max_retries):
        try:
            api = CoreAPI()
            api.initialize()
            
            # ... 执行流程 ...
            
            result = api.execute_workflow("workflow_name")
            
            if result.success:
                api.shutdown()
                return 0
            else:
                print(f"尝试 {attempt + 1} 失败: {result.errors}")
                
        except Exception as e:
            print(f"尝试 {attempt + 1} 出错: {e}")
        
        finally:
            if 'api' in locals():
                api.shutdown()
    
    print(f"所有 {max_retries} 次尝试都失败")
    return 1
```

### 6.2 并行处理
处理多个工作流：
```python
def process_multiple_workflows():
    """处理多个工作流"""
    api = CoreAPI()
    api.initialize()
    
    workflows = [
        "workflow1.json",
        "workflow2.json",
        "workflow3.json"
    ]
    
    results = []
    
    for workflow_path in workflows:
        print(f"\n处理工作流: {workflow_path}")
        
        # 为每个工作流创建独立的项目
        api.create_project(f"项目_{Path(workflow_path).stem}", f"./project_{Path(workflow_path).stem}")
        api.add_input_files(["data.xlsx"])
        api.add_template_files(["template.docx"])
        api.set_output_directory(f"./output_{Path(workflow_path).stem}")
        
        api.load_workflow(workflow_path)
        result = api.execute_workflow("workflow_name")
        results.append(result)
        
        # 保存项目
        api.save_project()
        api.close_project()
    
    api.shutdown()
    
    # 汇总结果
    success_count = sum(1 for r in results if r.success)
    print(f"\n完成: {success_count}/{len(workflows)} 个工作流成功")
    
    return 0
```

### 6.3 动态工作流生成
根据输入动态创建工作流：
```python
def create_dynamic_workflow(input_files):
    """根据输入文件动态创建工作流"""
    steps = []
    
    # 步骤1: 提取所有文件
    extract_step = {
        "step_id": "extract_all",
        "plugin_name": "excel_extractor",
        "config": {
            "PlaceHolders": []
        }
    }
    
    # 为每个输入文件添加提取配置
    for i, file_path in enumerate(input_files):
        extract_step["config"]["PlaceHolders"].append({
            "PlaceHolderName": f"data_{i}",
            "Sheet": ["Sheet1"],
            "Column": [""]
        })
    
    steps.append(extract_step)
    
    # 步骤2: 生成文档
    steps.append({
        "step_id": "generate",
        "plugin_name": "docx_replacer",
        "config": {
            "Target": ["template.docx"]
        }
    })
    
    workflow = {
        "name": "动态生成的工作流",
        "description": f"处理 {len(input_files)} 个文件",
        "version": "1.0.0",
        "steps": steps
    }
    
    # 保存工作流
    with open("dynamic_workflow.json", "w", encoding="utf-8") as f:
        json.dump(workflow, f, ensure_ascii=False, indent=2)
    
    return "dynamic_workflow.json"
```

### 7.2 单步执行工作流
```python
def debug_workflow():
    """调试工作流执行"""
    api = CoreAPI()
    api.initialize()
    
    # 加载工作流
    api.load_workflow("workflow.json")
    
    # 获取工作流定义
    workflow = api.workflow_engine.load_workflow("workflow.json")
    
    # 逐步执行
    for i, step in enumerate(workflow.steps):
        print(f"\n执行步骤 {i}: {step.step_id}")
        
        # 获取插件
        plugin = api.plugin_manager.get_plugin(step.plugin_name)
        
        # 执行单步
        result = plugin.execute(**step.config)
        
        print(f"结果: {result.success}")
        if result.data:
            print(f"数据字段: {list(result.data.keys())}")
        
        if not result.success:
            print(f"错误: {result.errors}")
            break
    
    api.shutdown()
```
